Finding the root of two 2 variants real functions.

Parameters
[in]fthe real function of 2 variants and 2 output.
[in,out]xthe initial guess, double array of 2 dimension, will be updated and converge to the root.
[in]tolthe precision of root.
[in]tolfthe error of function value at root.
[in]niterthe maximum iteration epochs.

Example file eg_root_impf_2f2_array_newton.c:

#include <stdio.h>
#include <math.h>
#include <impf/impf.h>
void f(const unsigned int m, const unsigned int n, const double * x, double * y)
{
y[0] = 1.5 + x[0] + x[1] + pow(0.5 + x[0], 2) + pow(0.5 + x[1], 3);
y[1] = 2.0 + x[0] + x[1] + pow(1.0 + x[1], 2) + pow(1.0 + x[0], 3);
}
int main(void)
{
FILE * out = fopen("eg_root_impf_2f2_array_newton.out", "w");
if(NULL == out) return -1;
double x[2] = {-0.0, -0.0}; /* the initial guess */
double y[2];
if(IMPF_EXIT_SUCCESS == impf_subrt_2f2_array_root_newton(f, x, 1e-8, 1e-8, 1000))
{
f(2, 2, x, y);
fprintf(out, "x = (%f, %f)\n", x[0], x[1]);
fprintf(out, "f(x) = (%f, %f)\n", y[0], y[1]);
}
else
{
fprintf(out, "Newton method fail to converge.\n");
}
fclose(out);
return 0;
}
